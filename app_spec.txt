<project_specification>
  <project_name>eSIM4Travel - Travel eSIM Provider</project_name>

  <overview>
    Build a fully functional mockup recreation of esim4travel.com - an eSIM provider offering affordable
    data packages for travelers across 190+ destinations worldwide. The application should provide a
    modern e-commerce experience for browsing destinations, selecting data plans, and completing purchases.
    The UI should be clean, travel-inspired, and mobile-first with excellent user experience using
    Tailwind CSS and React.
  </overview>

  <technology_stack>
    <frontend>
      <framework>React with Vite</framework>
      <styling>Tailwind CSS (via CDN)</styling>
      <state_management>React hooks and context</state_management>
      <routing>React Router for navigation</routing>
      <icons>Heroicons or Lucide React</icons>
      <port>Only launch on port {frontend_port}</port>
    </frontend>
    <backend>
      <runtime>Node.js with Express</runtime>
      <database>SQLite with better-sqlite3</database>
      <payment>Mock payment processing (no real transactions)</payment>
    </backend>
    <communication>
      <api>RESTful endpoints</api>
    </communication>
  </technology_stack>

  <key_interactions>
    <search_flow>
      1. User clicks search or types in hero search
      2. Autocomplete shows matching destinations
      3. User selects destination
      4. Navigate to destination detail page
    </search_flow>

    <purchase_flow>
      1. User browses destinations or searches
      2. Clicks on destination card
      3. Views available packages
      4. Clicks "Add to Cart" on desired package
      5. Cart drawer slides open
      6. User adjusts quantity if needed
      7. Optionally applies promo code
      8. Clicks "Checkout"
      9. Enters email address
      10. Selects payment method (mock)
      11. Reviews order and accepts terms
      12. Clicks "Place Order"
      13. Sees confirmation with QR code
    </purchase_flow>

    <cart_management>
      1. Add item: Cart drawer opens, item added
      2. Update quantity: Inline +/- buttons
      3. Remove item: Click X, confirm removal
      4. Apply promo: Enter code, see discount
      5. Proceed to checkout: Click button
    </cart_management>

    <support_flow>
      1. User clicks Support in nav
      2. Views installation guides by device
      3. Browses FAQ by category
      4. If needed, opens contact form
      5. Submits support request
      6. Sees confirmation message
    </support_flow>
